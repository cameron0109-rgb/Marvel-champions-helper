// --- Game Functions ---
function shuffle(array) {
  for (let i = array.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [array[i], array[j]] = [array[j], array[i]];
  }
}

function drawCard() {
  if (heroDeck.length === 0) return alert("Deck is empty!");
  let card = heroDeck.pop();
  heroHand.push(card);
  renderHand();
}

function renderHand() {
  handCardsEl.innerHTML = "";
  heroHand.forEach(c => {
    let img = document.createElement("img");
    img.src = c.image;
    img.alt = c.name;
    img.className = "card";
    handCardsEl.appendChild(img);
  });
  deckCountEl.innerText = heroDeck.length;
}

function advanceVillainStage() {
  villain.stage++;
  villainStage.innerText = villain.stage;
}

function spawnMinion() {
  villain.minions++;
  minionsCount.innerText = villain.minions;
}

// --- Init ---
shuffle(heroDeck);
deckCountEl.innerText = heroDeck.length;
renderHand();
</script>
